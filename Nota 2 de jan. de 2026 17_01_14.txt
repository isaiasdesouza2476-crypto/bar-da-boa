const SENHA = "1430";
let raspadinhasPermitidas = 0;

function selecionarValor(qtd) {
    raspadinhasPermitidas = qtd;
    document.getElementById("telaValores").classList.add("hidden");
    document.getElementById("telaSenha").classList.remove("hidden");
}

function validarSenha() {
    const s = document.getElementById("senha").value;
    if (s === SENHA) {
        document.getElementById("telaSenha").classList.add("hidden");
        iniciarJogo();
    } else {
        alert("Senha incorreta");
    }
}

function iniciarJogo() {
    document.getElementById("telaJogo").classList.remove("hidden");

    const total = 10;
    let premios = [
        "Vale 1 Cerveja",
        "Vale 1 Cerveja",
        "Vale 1 Cop√£o",
        "Vale 1 Cop√£o"
    ];

    let resultados = Array(total).fill("N√£o foi dessa vez");
    premios.forEach(p => {
        let i;
        do {
            i = Math.floor(Math.random() * total);
        } while (resultados[i] !== "N√£o foi dessa vez");
        resultados[i] = p;
    });

    const grid = document.getElementById("grid");
    grid.innerHTML = "";

    resultados.forEach(res => {
        const card = document.createElement("div");
        card.className = "card";
        card.textContent = "Raspar";

        card.onclick = () => {
            if (card.classList.contains("revelado")) return;
            if (raspadinhasPermitidas <= 0) return;

            raspadinhasPermitidas--;
            card.classList.add("revelado");

            if (res.includes("Vale")) {
                card.innerHTML = "üéâ<br><span class='premio'>" + res + "</span>";
            } else {
                card.innerHTML = "‚ùå<br><span class='perdeu'>" + res + "</span>";
            }
        };

        grid.appendChild(card);
    });
}
